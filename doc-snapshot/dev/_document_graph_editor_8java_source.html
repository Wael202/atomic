91  public void init(IEditorSite site, IEditorInput input) throws PartInitException {
92  if (!(input instanceof FileEditorInput)) {
93  log.error("Input for editors of type {} must be of type {}, but the provided input is of type {}!", this.getClass().getName(), FileEditorInput.class.getName(), input.getClass().getName(), new AtomicGeneralException());
94  }
95  else {
96  String filePath = ((FileEditorInput) input).getPath().makeAbsolute().toOSString();
97  log.trace("Input for editor {} is file '{}'.", this.getClass().getName(), filePath);
98  IProject iProject = ((FileEditorInput) input).getFile().getProject();
99  IFile saltProjectResource = null;
100  try {
101  for (IResource member : iProject.members()) {
102  if (member instanceof IFile && member.getName().equalsIgnoreCase("saltProject.salt")) { // FIXME Externalize and treat for case, etc
103  saltProjectResource = (IFile) member;
104  }
105  }
106  }
107  catch (CoreException e) {
108  // TODO Auto-generated catch block
109  e.printStackTrace();
110  }
111  String projectFilePath = saltProjectResource.getRawLocation().toOSString();
112  SaltProject saltProject = SaltFactory.createSaltProject();
113  saltProject.loadCorpusStructure(URI.createFileURI(projectFilePath));
114  for (SDocument doc : saltProject.getCorpusGraphs().get(0).getDocuments()) {
115  if (doc.getDocumentGraphLocation().equals(URI.createFileURI(filePath))) {
116  doc.loadDocumentGraph();
117  graph = doc.getDocumentGraph();
118  }
119  }
120  log.trace("Loaded document graph {}.", graph);
121  setSite(site);
122  setInput(input);
123 
124  // Set up editor for automatic context switches on activation/deactivation
125  PlatformUI.getWorkbench().getActiveWorkbenchWindow().getActivePage().addPartListener(new PartContextListener(site.getId(), site.getPluginId()));
126  }
127  }
