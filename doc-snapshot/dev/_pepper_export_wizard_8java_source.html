55  /*
56  * @copydoc @see org.eclipse.ui.IWorkbenchWizard#init(org.eclipse.ui.IWorkbench, org.eclipse.jface.viewers.IStructuredSelection)
57  */
58  @Override
59  public void init(IWorkbench workbench, IStructuredSelection selection) {
60  initialize();
61  if (1 == selection.size()) {
62  Object element = selection.getFirstElement();
63  if (element instanceof IProject) {
64  selectedProject = (IProject) element;
65  }
66  else {
67  new MessageDialog(this.getShell(), "Error", null, "Selection is not a project!", MessageDialog.ERROR, new String[] { IDialogConstants.OK_LABEL }, 0).open();
68  }
69  }
70  else {
71  new MessageDialog(this.getShell(), "Error", null, "To run the Pepper Export Wizard select exactly one project!", MessageDialog.ERROR, new String[] { IDialogConstants.OK_LABEL }, 0).open();
72  }
73  }
