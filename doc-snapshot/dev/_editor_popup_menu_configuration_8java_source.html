108  private void executeCreateTokenCommand(Boolean isNullToken) {
109  NatEventData data = (NatEventData) menu.getData(MenuItemProviders.NAT_EVENT_DATA_KEY);
110  NatTable natTable = data.getNatTable();
111  int columnPosition = data.getColumnPosition();
112  IUniqueIndexLayer bodyLayer = ((ViewportLayer) ((GridLayer) natTable.getLayer()).getBodyLayer()).getScrollableLayer();
113  int absoluteTokenIndex = LayerUtil.convertColumnPosition(natTable, columnPosition, bodyLayer);
114  List<SToken> sortedTokens = graph.getSortedTokenByText();
115  boolean isLastToken = (sortedTokens.size() - 1 == absoluteTokenIndex);
116  SToken selectedToken = sortedTokens.get(absoluteTokenIndex);
117 
118  ICommandService commandService = PlatformUI.getWorkbench().getActiveWorkbenchWindow().getService(ICommandService.class);
119  IHandlerService handlerService = PlatformUI.getWorkbench().getActiveWorkbenchWindow().getService(IHandlerService.class);
120 
121  // FIXME Externalize string
122  Command createTokenCmd = commandService.getCommand("org.corpus_tools.atomic.tokeneditor.commands.createToken");
123 
124  Event event = new Event();
125  event.data = selectedToken;
126  event.widget = natTable;
127 
128  Map<String, Object> params = new HashMap<>();
129  // FIXME Externalize strings
130  params.put("org.corpus_tools.atomic.tokeneditor.commands.createToken.isNullToken", isNullToken);
131  params.put("org.corpus_tools.atomic.tokeneditor.commands.createToken.isLastToken", isLastToken);
132  ParameterizedCommand parameterizedCmd = ParameterizedCommand.generateCommand(createTokenCmd, params);
133  try {
134  handlerService.executeCommand(parameterizedCmd, event);
135  }
136  catch (Exception e1) {
137  e1.printStackTrace();
