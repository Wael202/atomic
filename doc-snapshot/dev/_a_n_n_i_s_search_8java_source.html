183  private IFile searchDocumentInResource(IResource res, String documentName) throws CoreException {
184 
185  if(res instanceof IFile) {
186  IFile file = (IFile) res;
187  if((documentName + ".salt").equals(file.getName())) {
188  return file;

190  } else if (res instanceof IContainer ){

202  private void executeSearch(Composite parent, String aql) {
203 
204  lblStatus.setText("Searching...");
205  table.removeAll();
206 
207  for(int c=table.getColumnCount()-1; c >= 0; c--) {
208  table.getColumn(c).dispose();
209  }
210 
211  Job j = new Job("Searching in corpus") {
212 
213  @Override
214  protected IStatus run(IProgressMonitor monitor) {
215 
216  try {
217  MatchGroup result = search.find(aql);
218 
219  uiSync.asyncExec(() -> {
220  // find the maximal number of nodes per match and add a column for each
221  int maxNumNodes = 0;
222  for (Match m : result.getMatches()) {
223  maxNumNodes = Math.max(maxNumNodes, m.getSaltIDs().size());
224  }
225 
226  // add columns for corpus and document name
227  TableColumn corpusColumn = new TableColumn(table, SWT.NULL);
228  corpusColumn.setText("corpus");
229 
230  TableColumn documentColumn = new TableColumn(table, SWT.NULL);
231  documentColumn.setText("document");
232 
233  for (int i = 1; i <= maxNumNodes; i++) {
234  TableColumn c = new TableColumn(table, SWT.NULL);
235  c.setText("node #" + i);
236  }
237 
238  int displayMatchCount = 0;
239  for (Match m : result.getMatches()) {
240 
241  TableItem item = new TableItem(table, SWT.NULL);
242 
243  item.setData(m);
244 
245  if(!m.getSaltIDs().isEmpty()) {
246  List<String> path = pathSplitter.splitToList(
247  m.getSaltIDs().iterator().next().getPath());
248  item.setText(0, path.get(0)); // corpus
249  item.setText(1, path.get(path.size()-1)); // document
250  } else {
251  item.setText(0, "<unknown>");
252  item.setText(1, "<unknown>");
253  }
254 
255  int nodeIdx = 0;
256  for (URI u : m.getSaltIDs()) {
257  item.setText(2+nodeIdx, u.getFragment());
258  nodeIdx++;
259  }
260 
261 
262  displayMatchCount++;
263  if(displayMatchCount > 10000) {
264  break;
265  }
266  }
267 
268  for (int i = 0; i < table.getColumnCount(); i++) {
269  table.getColumn(i).pack();
270  }
271  long matchCount = result.getMatches().size();
272  if(matchCount > 10000) {
273  lblStatus.setText("Found " + matchCount + " matches. (Only displaying first 10.000 matches)");
274  } else {
275  lblStatus.setText("Found " + matchCount + " matches.");
276  }
277  });
278 
279  } catch(AnnisQLSyntaxException | AnnisQLSemanticsException ex ) {
280  uiSync.asyncExec(() -> {
281  lblStatus.setText("ERROR: " + ex.getMessage());
282  MessageDialog.openError(parent.getShell(), "Can't parse AQL", ex.getMessage());
283  });
284  }
285  return Status.OK_STATUS;
286  }
287 
288  };
289  j.setUser(true);
290  j.setPriority(Job.LONG);
291  j.schedule();
292  }

302  private void openMatchInEditor(Widget selected) {
303 
304  if(selected instanceof TableItem) {
305  TableItem selectedItem = (TableItem) selected;
306  if(selectedItem.getData() instanceof Match) {
307  Match match = (Match) selectedItem.getData();
308  // get first match (which already contains the corpus name and document name)
309  List<String> path = pathSplitter.splitToList(match.getSaltIDs().get(0).getPath());
310  // find the document in the Workspace
311  IWorkspace workspace = ResourcesPlugin.getWorkspace();
312  IWorkspaceRoot root = workspace.getRoot();
313 
314  for(IProject p : root.getProjects()) {
315  if(path.get(0).equals(p.getName())) {
316  try {
317  IFile matchingDoc = searchDocumentInResource(p, path.get(path.size()-1));
318  if(matchingDoc != null) {
319 
320  IEditorPart newEditor = null;
321  if(editorDesc == null) {
322  // use the default editor
323  IEditorDescriptor desc = PlatformUI.getWorkbench().getEditorRegistry().getDefaultEditor(matchingDoc.getName());
324  newEditor = page.openEditor(new FileEditorInput(matchingDoc), desc.getId());
325  } else {
326  newEditor = page.openEditor(new FileEditorInput(matchingDoc), editorDesc.getId());
327 
328  }
329 
330  if(newEditor instanceof SaltNodeSelectable) {
331  List<String> selectedNodeNames = new LinkedList<>();
332  for(URI uri : match.getSaltIDs()) {
333  selectedNodeNames.add(uri.toASCIIString());
334  }
335  ((SaltNodeSelectable) newEditor).setSelection(selectedNodeNames);
336  }
337  }
338 
339  } catch (CoreException ex) {
340  log.error("Could not open match in editor", ex);
341  }
342  break;
343  }
344  }
345  }
346  }
