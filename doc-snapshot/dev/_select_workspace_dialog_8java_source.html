136  private String suggestWorkspaceDirectory() {
137  StringBuffer buffer = new StringBuffer();
138  String userHome = System.getProperty("user.home");
139  log.trace("Value for \"user.home\" system property is \"{}\".", userHome);
140  buffer.append(userHome);
141  buffer.append(File.separator);
142  buffer.append(ATOMIC_WORKSPACE_DIRECTORY_DEFAULT_NAME);
143  log.info("Suggesting workspace directory to be \"atomic-workspace\" in the user's home directory ({}).", userHome);
144  return buffer.toString();

169  private void createBrowseRow(Composite composite) {
170  Composite row = new Composite(composite, SWT.NONE);
171  GridLayout layout = new GridLayout(3, false);
172  row.setLayout(layout);
173 
174  CLabel label = new CLabel(row, SWT.NONE);
175  label.setText("Workspace path:");
176 
177  workspacePathCombo = new Combo(row, SWT.BORDER);
178  String workspaceRoot = preferences.get(PREF_KEY_WORKSPACE_ROOT, "");
179  if (workspaceRoot == null || workspaceRoot.length() == 0) {
180  workspaceRoot = suggestWorkspaceDirectory();
181  }
182  workspacePathCombo.setText(workspaceRoot == null ? "" : workspaceRoot);
183 
184  String lastUsedWorkspace = preferences.get(PREF_KEY_LAST_WORKSPACES, "");
185  lastWorkspacesList = new ArrayList<String>();
186  if (lastUsedWorkspace != null) {
187  String[] allWorkspaces = lastUsedWorkspace.split(SPLIT_CHAR);
188  for (String oneOfAllWorkspace : allWorkspaces)
189  lastWorkspacesList.add(oneOfAllWorkspace);
190  }
191  for (String lastOfAllWorkspace : lastWorkspacesList)
192  workspacePathCombo.add(lastOfAllWorkspace);
193 
194  Button browseButton = new Button(row, SWT.PUSH);
195  browseButton.setText("&Browse...");
196  browseButton.addListener(SWT.Selection, new Listener() {
197  @Override
198  public void handleEvent(Event event) {
199  DirectoryDialog directoryDialog = new DirectoryDialog(getParentShell());
200  directoryDialog.setText("Select Workspace Root");
201  directoryDialog.setMessage(MESSAGE_INFO);
202  directoryDialog.setFilterPath(workspacePathCombo.getText());
203  String pick = directoryDialog.open();
204  if (pick == null && workspacePathCombo.getText().length() == 0) {
205  setMessage(MESSAGE_ERROR, IMessageProvider.ERROR);
206  }
207  else {
208  setMessage(MESSAGE_DEFAULT);
209  if (pick != null) {
210  workspacePathCombo.setText(pick);
211  }
212  else {
213  log.trace("Variable \'pick\' is null.");
214  }
215  }
216  }

220  /*
221  * @see org.eclipse.jface.dialogs.Dialog#okPressed()
222  */
223  @Override
224  protected void okPressed() {
225  String workspacePathString = workspacePathCombo.getText();
226 
227  if (workspacePathString.length() == 0) {
228  setMessage(MESSAGE_ERROR, IMessageProvider.ERROR);
229  return;
230  }
231 
232  String workspaceCheckResult = checkWorkspaceDirectory(getParentShell(), workspacePathString, true, true);
233  if (workspaceCheckResult != null) {
234  setMessage(workspaceCheckResult, IMessageProvider.ERROR);
235  return;
236  }
237 
238  lastWorkspacesList.remove(workspacePathString);
239  if (!lastWorkspacesList.contains(workspacePathString)) {
240  lastWorkspacesList.add(0, workspacePathString);
241  }
242 
243  if (lastWorkspacesList.size() > MAX_WORKSPACES_IN_HISTORY) {
244  List<String> workspaceToBeRemovedFromList = new ArrayList<String>();
245  for (int i = MAX_WORKSPACES_IN_HISTORY; i < lastWorkspacesList.size(); i++) {
246  workspaceToBeRemovedFromList.add(lastWorkspacesList.get(i));
247  }
248  lastWorkspacesList.removeAll(workspaceToBeRemovedFromList);
249  }
250 
251  // Build a list of previously used workspaces
252  StringBuffer previouslyUsedWorkspaces = new StringBuffer();
253  for (int i = 0; i < lastWorkspacesList.size(); i++) {
254  previouslyUsedWorkspaces.append(lastWorkspacesList.get(i));
255  if (i != lastWorkspacesList.size() - 1) {
256  previouslyUsedWorkspaces.append(SPLIT_CHAR);
257  }
258  }
259 
260  // Store remember-workspace value & previous workspaces in preferences
261  preferences.putBoolean(PREF_KEY_REMEMBER_WORKSPACE, rememberWorkspaceButton.getSelection());
262  preferences.put(PREF_KEY_LAST_WORKSPACES, previouslyUsedWorkspaces.toString());
263 
264  // Create the workspace
265  boolean workspaceCreated = checkAndCreateWorkspaceRoot(workspacePathString);
266  if (!workspaceCreated) {
267  setMessage("The workspace could not be created, please check the error log");
268  return;
269  }
270 
271  // Save the workspace path as String
272  selectedWorkspaceLocationAsString = workspacePathString;
273  preferences.put(PREF_KEY_WORKSPACE_ROOT, workspacePathString);
274  super.okPressed();
275  }

286  public static String checkWorkspaceDirectory(Shell parentShell, String workspaceLocation, boolean askForCreation, boolean calledFromWorkspaceDialog) {
287  File workspaceFile = new File(workspaceLocation + File.separator + WORKSPACE_IDENTIFIER_FILE_NAME);
288 
289  File workspaceDirectory = new File(workspaceLocation);
290  if (!workspaceDirectory.exists()) {
291  if (askForCreation) {
292  boolean create = MessageDialog.openConfirm(parentShell, "New workspace directory", "The directory does not exist. Would you like to create it?");
293  if (create) {
294  try {
295  log.info("Creating directories and workspace file in {}.", workspaceDirectory);
296  workspaceDirectory.mkdirs();
297  workspaceFile.createNewFile();
298  }
299  catch (Exception err) {
300  return "Error creating directories, please check folder permissions!";
301  }
302  }
303 
304  if (!workspaceDirectory.exists()) {
305  return "The selected directory does not exist!";
306  }
307  }
308  }
309 
310  if (!workspaceDirectory.canRead()) {
311  return "The selected directory is not readable!";
312  }
313 
314  if (!workspaceDirectory.isDirectory()) {
315  return "The selected path is not a directory!";
316  }
317 
318  if (calledFromWorkspaceDialog) {
319  if (!workspaceFile.exists()) {
320  boolean create = MessageDialog.openConfirm(parentShell, "New workspace", "The directory '" + workspaceFile.getAbsolutePath() + "' will be transformed into a workspace. Workspace-specific files and directories will be created.\n\nWould you like to create a workspace in the selected location?");
321  if (create) {
322  try {
323  log.info("Creating directories and workspace file in {}.", workspaceDirectory);
324  workspaceDirectory.mkdirs();
325  workspaceFile.createNewFile();
326  }
327  catch (Exception err) {
328  return "Error creating directories, please check folder permissions";
329  }
330  }
331  else {
332  return "Please select a directory for your workspace";
333  }
334 
335  if (!workspaceFile.exists()) {
336  return "The selected directory does not exist";
337  }
338 
339  return null;
340  }
341  }
342  else {
343  if (!workspaceFile.exists()) {
344  return "The selected directory is not a workspace directory";
345  }
346  }
347 
348  return null;
349  }

357  public static boolean checkAndCreateWorkspaceRoot(String workspaceDirectory) {
358  try {
359  File workspaceDirectoryFile = new File(workspaceDirectory);
360  if (!workspaceDirectoryFile.exists())
361  return false;
362 
363  File workspaceIdentifierFile = new File(workspaceDirectory + File.separator + WORKSPACE_IDENTIFIER_FILE_NAME);
364  if (!workspaceIdentifierFile.exists() && !workspaceIdentifierFile.createNewFile()) {
365  return false;
366  }
367  return true;
368  }
369  catch (Exception e) {
370  log.error("There was an error during checking and/or creating the Atomic workspace at given location.", e);
371  return false;
372  }
373  }
