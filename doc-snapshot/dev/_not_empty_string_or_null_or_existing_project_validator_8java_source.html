47  public NotEmptyStringOrNullOrExistingProjectValidator(String fieldName) {
48  this.fieldName = fieldName;

66  public IStatus validate(Object value) {
67  if (value instanceof String) {
68  String text = (String) value;
69  if (text == null || text.isEmpty()) {
70  return ValidationStatus.error(fieldName + " must not be empty!");
71  }
72  else if (existingProjects.contains(text)) {
73  return ValidationStatus.error("A project with the name \"" + text + "\" already exists in the workspace!");
74  }
75  else {
76  return ValidationStatus.ok();
77  }
