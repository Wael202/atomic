79  private void syncTextPosition(NatTable natTable) {
80  SelectionLayer selectionLayer = null;
81  ILayer underlyingLayer = natTable.getLayer();
82  if (underlyingLayer instanceof GridLayer) {
83  ILayer bodyLayer = ((GridLayer) underlyingLayer).getBodyLayer();
84  if (bodyLayer instanceof ViewportLayer) {
85  IUniqueIndexLayer scrollLayer = ((ViewportLayer) bodyLayer).getScrollableLayer();
86  if (scrollLayer instanceof SelectionLayer) {
87  selectionLayer = (SelectionLayer) scrollLayer;
88  }
89  int selectedCol = selectionLayer.getSelectedColumnPositions()[0];
90  Object pos = selectionLayer.getDataValueByPosition(selectedCol, 1); // FIXME: Get int via constant in editor so that it cannot break if row number changes
91  if (pos instanceof String) {
92  String[] split = ((String) pos).split("\\s-\\s", 2);
93  int start = Integer.parseInt(split[0].trim());
94  int end = Integer.parseInt(split[1].trim());
95  text.setCaretOffset(start);
96  text.setSelection(start, end);
97  }
98  }
99  }
