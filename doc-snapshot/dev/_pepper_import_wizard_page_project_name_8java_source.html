88  protected void updatePageComplete() {
89  String projectName = getProjectName();
90  if (projectName != null) {
91  if (ResourcesPlugin.getWorkspace().getRoot().getProject(getProjectName()).exists()) {
92  setMessage(null);
93  setErrorMessage("Project does already exist - choose another name.");
94  setPageComplete(false);

96  else {
97  setMessage(null);

102  else {

105  setPageComplete(false);
106  }
