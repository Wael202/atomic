200  private Set<Integer> getSelectedSegmentIdxForNodes(List<String> nodeIDs) {
201  // sort the available segments by their size (smallest first)
202  List<Integer> segmentIdx = new LinkedList<>();
203  for(int i=0; i < textRangeTable.getItemCount(); i++) {
204  segmentIdx.add(i);
205  }
206  segmentIdx.sort(new Comparator<Integer>() {
207  @Override
208  public int compare(Integer o1, Integer o2) {
209  TableItem item1 = textRangeTable.getItem(o1);
210  TableItem item2 = textRangeTable.getItem(o2);
211 
212  Range<Long> range1 = (Range<Long>) item1.getData("range");
213  Range<Long> range2 = (Range<Long>) item2.getData("range");
214 
215  return ComparisonChain.start()
216  .compare(range1.upperEndpoint() - range1.lowerEndpoint(), range2.upperEndpoint() - range2.lowerEndpoint())
217  .result();
218  }
219  });
220 
221  Set<Integer> selectedIdx = new HashSet<>();
222 
223  // find all segments that include the nodes
224  for(String id : nodeIDs) {
225  SNode n = getGraph().getNode(id);
226  if(n != null) {
227 
228  List<DataSourceSequence> overlappedDS = graph.getOverlappedDataSourceSequence(n,
229  SALT_TYPE.STEXT_OVERLAPPING_RELATION);
230  if (overlappedDS != null) {
231  for (DataSourceSequence seq : overlappedDS) {
232  if (seq.getDataSource() instanceof STextualDS) {
233  Range<Long> selectedNodeRange = Range.closedOpen(seq.getStart().longValue(), seq.getEnd().longValue());
234 
235  for(int idx : segmentIdx) {
236  Range<Long> rangeSegment = (Range<Long>) textRangeTable.getItem(idx).getData("range");
237  if(rangeSegment.encloses(selectedNodeRange)) {
238  selectedIdx.add(idx);
239  break;

241  }

244  }
245  }
246  }
247  }

395  private void calculateSegments(SDocumentGraph graph) {
396  textRangeTable.removeAll();
397 
398  ExportFilter currentFilter = new RootFilter();
399 
400  Multimap<STextualDS, Range<Long>> sortedDS = TreeMultimap.create(new STextualDSComparator(),
401  new RangeStartComparator<>());
402 
403  List<SNode> roots = graph.getRoots();
404  if (roots != null) {
405  for (SNode r : roots) {
406 
407  if (currentFilter.includeNode(r)) {
408  List<DataSourceSequence> overlappedDS = graph.getOverlappedDataSourceSequence(r,
409  SALT_TYPE.STEXT_OVERLAPPING_RELATION);
410  if (overlappedDS != null) {
411  for (DataSourceSequence seq : overlappedDS) {
412  if (seq.getDataSource() instanceof STextualDS) {
413  sortedDS.put((STextualDS) seq.getDataSource(),
414  Range.closedOpen(seq.getStart().longValue(), seq.getEnd().longValue()));
415 
416  }
417  }
418  }
419  }
420  }
421  }
422 
423  for (Map.Entry<STextualDS, Range<Long>> e : sortedDS.entries()) {
424  TableItem item = new TableItem(textRangeTable, SWT.NONE);
425  item.setText(
426  e.getKey().getName() + ": " + e.getValue().lowerEndpoint() + ".." + e.getValue().upperEndpoint());
427  item.setData("range", e.getValue());
428  item.setData("text", e.getKey());
429  }

467  private class Filter implements ExportFilter {
468 
469  @Override
470  public boolean includeNode(SNode node) {
471 
472  boolean include = false;
473 
474  // check if the node covers a currently selected range
475  @SuppressWarnings("rawtypes")
476  List<DataSourceSequence> seqList = getGraph().getOverlappedDataSourceSequence(node,
477  SALT_TYPE.STEXT_OVERLAPPING_RELATION);
478  if (seqList != null) {
479  outerLoop: for (DataSourceSequence seq : seqList) {
480 
481  Range<Long> nodeRange = Range.closedOpen(seq.getStart().longValue(), seq.getEnd().longValue());
482 
483  for (TableItem item : textRangeTable.getSelection()) {
484 
485  Range<Long> itemRange = (Range<Long>) item.getData("range");
486  STextualDS itemText = (STextualDS) item.getData("text");
487 
488  if (itemText == seq.getDataSource() && nodeRange.isConnected(itemRange)) {
489  include = true;
490  break outerLoop;
491  }
492 
493  }
494 
495  }
496  }
497 
498  if (node instanceof SSpan) {
499  include = include && btnIncludeSpans.getSelection();
500  }
501 
502  // additionally check for valid annotation
503  if (include && !txtSegmentFilter.getText().isEmpty() && !(node instanceof SToken)) {
504  if (node.getAnnotations() != null) {
505  boolean annoFound = false;
506  for (SAnnotation anno : node.getAnnotations()) {
507  if (anno.getName().contains(txtSegmentFilter.getText())) {
508  annoFound = true;
509  break;
510  }
511  }
512  include = annoFound;
513  }
514  }
515 
516  return include;
517  }
