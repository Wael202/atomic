76  @Override
77  public boolean performFinish() { // FIXME: Add logs
78  AtomicPepperStarter pepperStarter = new AtomicPepperStarter();
79  pepperStarter.startPepper();
80  this.pepper = pepperStarter.getPepper();
81  log.trace("Procured an instance of {}: {}.", this.pepper.getClass().getName(), this.pepper.toString());
82  String pepperJobId = pepper.createJob();
83  PepperJob pepperJob = pepper.getJob(pepperJobId);
84  log.trace("Procured an instance of {}: {}.", pepperJob.getClass().getName(), pepperJob.toString());
85 
86  // Create import step
87  StepDesc importStepDesc = pepperJob.createStepDesc();
88  importStepDesc.setModuleType(MODULE_TYPE.IMPORTER);
89  String importPath = page.getCorpus().getPath();
90  importStepDesc.getCorpusDesc().setCorpusPath(URI.createFileURI(importPath));
91  log.trace("Procured an instance of {}: {}.", importStepDesc.getClass().getName(), importStepDesc.toString());
92  PepperModuleDesc importer = null;
93  for (PepperModuleDesc md : pepper.getRegisteredModules()) {
94  if (md.getModuleType().equals(MODULE_TYPE.IMPORTER) && md.getName().equals("TextImporter")) {
95  importer = md;
96  break;
97  }
98  }
99  importStepDesc.setName(importer.getName());
100  pepperJob.addStepDesc(importStepDesc);
101  log.info("Successfully set up an importer of type {} for the creation of a new Atomic project from path {}.", importer.getName(), importPath);
102 
103  // Set up Eclipse project
104  String projectName = page.getCorpus().getName();
105  IWorkspaceRoot workspaceRoot = ResourcesPlugin.getWorkspace().getRoot();
106  log.trace("Workspace root is {}.", workspaceRoot.getLocation().toOSString());
107  IProject iProject = workspaceRoot.getProject(projectName);
108  log.trace("Created IProject OBJECT with name \"{}\" at workspace root.", iProject.getName());
109  try {
110  iProject.create(null);
111  log.info("Successfully created Atomic project \"{}\".", iProject.getName());
112  iProject.open(null);
113  log.trace("Opened IProject \"{}\".", iProject.getName());
114  }
115  catch (CoreException e) {
116  MessageDialog.openError(Display.getCurrent() != null ? Display.getCurrent().getActiveShell() : Display.getDefault().getActiveShell(), "Atomic project note created!", "A project by the name of \"" + projectName + "\" already exists.");
117  log.error("Could not create and open the IProject for SaltProject {}!", projectName, e);
118  return false;
119  }
120  File iProjectLocation = new File(iProject.getLocation().toString());
121 
122  // Create export step
123  StepDesc exportStepDesc = pepperJob.createStepDesc();
124  exportStepDesc.setModuleType(MODULE_TYPE.EXPORTER);
125  String outputCanonicalPath;
126  try {
127  outputCanonicalPath = iProjectLocation.getCanonicalPath();
128  }
129  catch (IOException ex) {
130  outputCanonicalPath = iProjectLocation.getAbsolutePath();
131  }
132  exportStepDesc.getCorpusDesc().setCorpusPath(URI.createFileURI(outputCanonicalPath));
133  log.trace("Procured an instance of {}: {}.", importStepDesc.getClass().getName(), outputCanonicalPath);
134  PepperModuleDesc exporter = null;
135  for (PepperModuleDesc md : pepper.getRegisteredModules()) {
136  if (md.getModuleType().equals(MODULE_TYPE.EXPORTER) && md.getName().equals("SaltXMLExporter")) {
137  exporter = md;
138  break;
139  }
140  }
141  exportStepDesc.setName(exporter.getName());
142  pepperJob.addStepDesc(exportStepDesc);
143  log.info("Successfully set up an exporter of type {} for the creation of a new Atomic project to path {}.", importer.getName(), outputCanonicalPath);
144 
145  // Convert
146  log.trace("Starting conversion with Pepper job {}.", pepperJob.toString());
147  // TODO Add observation and reporting!
148  try {
149  pepperJob.convert();
150  } catch (Exception e) {
151  log.error("An error occurred during the conversion of a text corpus to a Salt project during the creation of a new project!", e);
152  } finally {
153  pepper.removeJob(pepperJobId);
154  }
155  try {
156  iProject.refreshLocal(IProject.DEPTH_INFINITE, null);
157  }
158  catch (CoreException e) {
159  log.error("Could not refresh IProject for SaltProject {}!", projectName, e);
160  }
161  // Finally, return true (i.e., wizard is done).
162  return true;
163  }
