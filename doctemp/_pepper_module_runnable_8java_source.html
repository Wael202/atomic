103  * @copydoc @see org.eclipse.jface.operation.IRunnableWithProgress#run(org.eclipse.core.runtime.IProgressMonitor)
104  */
105  @Override
106  public void run(final IProgressMonitor monitor) throws InvocationTargetException, InterruptedException {
107  try {
108  synchronized (cancelLock) {
109  // Check if run has been cancelled before start
110  if (cancelled) {
111  throw new InterruptedException();
112  }
113  else {
114  controlThread = Thread.currentThread();
115  // Thread cancellable via input in the ProgressMonitorDialog
116  moduleThread = new Thread("Pepper Module Thread #" + threadCounter.incrementAndGet()) {
117  @Override
118  public void run() {
119  try {
120  runPepper();
121  }
122  catch (IOException X) {
123  throw new RuntimeException(X);
124  }
125  catch (CoreException X) {
126  throw new RuntimeException(X);
127  }
128  }
129  };
130  }
131  }
132 
133  // Asynchronously monitor ProgressMonitor for cancellations
134  ScheduledFuture<?> cancellationCheck;
135  if (cancelable) {
136  // Monitoring thread
137  cancellationCheck = Executors.newSingleThreadScheduledExecutor().scheduleWithFixedDelay(new Runnable() {
138  @Override
139  public void run() {
140  if (monitor.isCanceled()) {
141  cancel(true);
142  }
143  }
144  }, 500, 500, TimeUnit.MILLISECONDS);
145  }
146  else {
147  cancellationCheck = null;
148  }
149 
150  // Start monitor
151  monitor.beginTask("Running ...", IProgressMonitor.UNKNOWN);
152  outcome = Boolean.FALSE;
153  try {
154  // Run module
155  moduleThread.start();
156 
157  Display display = Display.findDisplay(Thread.currentThread());
158  if (display == null) {
159  moduleThread.join();
160  }
161  else {
162  // Current thread is the UI Thread
163  while (!isCancelled() && moduleThread.isAlive()) {
164  if (controlThread.isInterrupted()) {
165  throw new InterruptedException();
166  }
167 
168  if (!display.readAndDispatch()) {
169  display.sleep();
170  }
171  }
172  }
173 
174  outcome = Boolean.TRUE;
175  }
176  finally {
177  // Stop monitor
178  monitor.done();
179 
180  // Stop monitor thread
181  if (cancellationCheck != null) {
182  cancellationCheck.cancel(true);
183  }
184 
185  // Signal cancellation
186  if (Thread.currentThread().isInterrupted()) {
187  throw new InterruptedException();
188  }
189  }
190  }
191  catch (InterruptedException X) {
192  new MessageDialog(Display.getCurrent().getActiveShell(), "Error", null, "Some error occured while running Pepper and unfortunately Pepper is not stoppable anymore. Letting it run wild.", MessageDialog.ERROR, new String[] { IDialogConstants.OK_LABEL }, 0).open();
193  }
194  catch (Throwable T) {
195  throw new InvocationTargetException(throwable = T);
196  }
197  finally {
198  moduleThread = null;
199  controlThread = null;
200  done = true;
201  semaphore.release(Integer.MAX_VALUE);
202  }
203  }

209  public boolean cancel(boolean mayInterruptIfRunning) {
210  if (cancelable) {
211  synchronized (cancelLock) {
212  if (!cancelled) {
213  Thread thread = controlThread;
214  if (thread != null) {
215  if (mayInterruptIfRunning) {
216  thread.interrupt();
217  cancelled = true;
218  }
219  }
220  else {
221  cancelled = true;
222  }
223  }
224 
225  return cancelled;
226  }
227  }
228  else {
229  return false;
230  }
231  }
