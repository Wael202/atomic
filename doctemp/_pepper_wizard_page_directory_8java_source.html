191  protected String validateTargetPath(File targetPath, boolean isFile) {
192  if (pepperWizard.getWizardMode() == WizardMode.EXPORT) {
193  if (targetPath.exists()) {
194  return isFile && targetPath.isFile() ? null : isFile ? "Target path exists but is not a file!" : targetPath.isDirectory() ? null : "Target path exists but is not a directory!";
195  }
196  else {
197  return null;
198  }
199  }
200  else {
201  if (isFile) {
202  return !targetPath.exists() ? "Target file does not exists!" : !targetPath.isFile() ? "Target path is not a file!" : !targetPath.canRead() ? "Target file can not be read!" : null;
203  }
204  else {
205  if (targetPath.exists()) {
206  if (targetPath.isDirectory()) {
207  File[] files = targetPath.listFiles();
208 
209  if (files != null && 0 < files.length) {
210  return null;
211  }
212  else {
213  return "Target directory contains no files!";
214  }
215  }
216  else {
217  return "Target path is not a directory!";
218  }
219  }
220  else {
221  return "Target directory does not exists!";
222  }
223  }
224  }
225  }
