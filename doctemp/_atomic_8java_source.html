60  return IApplication.EXIT_OK;

88  private boolean checkWorkspaceSetup(Display display) {
89  boolean isWorkspaceRemembered = SelectWorkspaceDialog.isRememberWorkspace();
90  String lastUsedWorkspace = SelectWorkspaceDialog.getLastWorkspace();
91  if (isWorkspaceRemembered && (lastUsedWorkspace == null || lastUsedWorkspace.length() == 0)) {
92  isWorkspaceRemembered = false;
93  }
94  if (isWorkspaceRemembered) {
95  String ret = SelectWorkspaceDialog.checkWorkspaceDirectory(Display.getDefault().getActiveShell(), lastUsedWorkspace, false, false);
96  if (ret != null) {
97  isWorkspaceRemembered = false;
98  }
99  }
100  log.trace("Remembered workspace: {}", isWorkspaceRemembered);
101  if (!isWorkspaceRemembered) {
102  SelectWorkspaceDialog workspaceDialog = new SelectWorkspaceDialog(false);
103  int retVal = workspaceDialog.open();
104  if (retVal == Window.CANCEL) {
105  if (workspaceDialog.getSelectedWorkspaceLocationAsString() == null) {
106  MessageDialog.openError(display.getActiveShell(), "Error", "Atomic can not start without a workspace and will now exit.");
107  log.info("No workspace set, exit application.");
108  return true;
109  }
110  }
111  else {
112  try {
113  instanceLocation.set(new URL("file", null, workspaceDialog.getSelectedWorkspaceLocationAsString()), false);
114  }
115  catch (IllegalStateException | IOException e) {
116  log.error("Setting the instance location didn't succeed!", e);
117  }
118  }
119  }
120  else {
121  // set the last used location and continue
122  try {
123  instanceLocation.set(new URL("file", null, lastUsedWorkspace), false);
124  }
125  catch (IllegalStateException | IOException e) {
126  log.error("Setting the instance location to the workspace last in use failed!", e);
127  }
128  }
129  return false;
130  }

140  return;

142  final IWorkbench workbench = PlatformUI.getWorkbench();
143  final Display display = workbench.getDisplay();
144  display.syncExec(new Runnable() {
145  public void run() {
146  if (!display.isDisposed())
147  log.trace("Closing workbench.");
148  workbench.close();
149  }
150  });
151  }
