108  private String getSourceTextFilesFromDialog(boolean isDirDialog) {
109  String outputCanonicalPath = null;
110  File file = null;
111  if (isDirDialog) {
112  DirectoryDialog dialog = new DirectoryDialog(Display.getCurrent() != null ? Display.getCurrent().getActiveShell() : Display.getDefault().getActiveShell(), SWT.OPEN);
113  if (dialog.open() != null) {
114  String fileName = dialog.getFilterPath();
115  file = new File(fileName);
116  }
117  }
118  else {
119  FileDialog dialog = new FileDialog(Display.getCurrent() != null ? Display.getCurrent().getActiveShell() : Display.getDefault().getActiveShell(), SWT.OPEN | SWT.SINGLE);
120  dialog.setFilterExtensions(new String[] { "*.txt" });
121  if (dialog.open() != null) {
122  String fileName = dialog.getFileName();
123  String filterPath = dialog.getFilterPath();
124  if (filterPath != null && filterPath.trim().length() > 0) {
125  file = new File(filterPath, fileName);
126  }
127  else {
128  file = new File(fileName);
129  }
130  }
131  }
132  try {
133  outputCanonicalPath = file.getCanonicalPath();
134  }
135  catch (IOException ex) {
136  outputCanonicalPath = file.getAbsolutePath();
137  }
138  return outputCanonicalPath;
139  }
