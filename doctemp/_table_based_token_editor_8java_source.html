133  @Override
134  public void keyPressed(KeyEvent e) {
135  /*
136  * CTRL + G selects the column in the table that has
137  * the same index as the token in the graph's list
138  * of ordered tokens by text whose overlapped text
139  * (inclusively) contains the current caret offset.
140  */
141  if ((e.stateMask & SWT.CTRL) != 0 && (e.keyCode == 'g' || e.keyCode == 't')) {
142  int index = -1;
143  int offset = text.getCaretOffset();
144  // Find the STextualRelation for the offset
145  relations:
146  for (STextualRelation rel : graph.getTextualRelations()) {
147  // TODO: Benchmark this vs. single if with <= and >= (3rd if here)
148  if (rel.getStart() == offset || rel.getEnd() == offset) {
149  index = graph.getSortedTokenByText().indexOf(rel.getSource());
150  break relations;
151  }
152  else if (rel.getEnd() > offset) {
153  if (rel.getStart() < offset) {
154  index = graph.getSortedTokenByText().indexOf(rel.getSource());
155  break relations;
156  }
157  }
158  }
159  if (index != -1) {
160  int pos = selectionLayer.getColumnPositionByIndex(index);
161  selectionLayer.selectColumn(pos, 0, false, false);
162  }
163  /*
164  * CTRL + T selects the table control.
165  */
166  if (e.keyCode == 't') {
167  natTable.setFocus();
168  }
169  }
170 
171  }
172  });
