107  public List<String> getDropInPaths() {
108  String rawList = this.getProperty(PepperStarterConfiguration.PROP_DROPIN_PATHS);
109  if (rawList != null) {
110  Iterator<String> it
111  = Splitter.on(',').trimResults().omitEmptyStrings().split(rawList).iterator();
112  List<String> result = new ArrayList<>();
113  while (it.hasNext()) {
114  result.add(it.next());
115  }
116  return result;
117  }
118  return null;
119  }
