76  private void jumpTo(NatTable natTable, int distance) {
77  SelectionLayer selectionLayer = null;
78  ILayer underlyingLayer = natTable.getLayer();
79  if (underlyingLayer instanceof GridLayer) {
80  ILayer bodyLayer = ((GridLayer) underlyingLayer).getBodyLayer();
81  if (bodyLayer instanceof ViewportLayer) {
82  IUniqueIndexLayer scrollLayer = ((ViewportLayer) bodyLayer).getScrollableLayer();
83  if (scrollLayer instanceof SelectionLayer) {
84  selectionLayer = (SelectionLayer) scrollLayer;
85  }
86  try {
87  int selectedCol = selectionLayer.getSelectedColumnPositions()[0];
88  int lastItemIndex = selectionLayer.getColumnCount() - 1;
89  int jumpPosition = selectedCol + distance;
90  if (jumpPosition > lastItemIndex) {
91  jumpPosition = lastItemIndex;
92  }
93  else if (jumpPosition < 0) {
94  jumpPosition = 0;
95  }
96  selectionLayer.selectColumn(jumpPosition, 0, false, false);
97  }
98  catch (ArrayIndexOutOfBoundsException e) {
99  // Ignore
100  }
101  }
102  }
