96  SubMonitor monitorDelete = SubMonitor.convert(monitor, root.getProjects().length);
97  monitorDelete.setTaskName("Deleting old corpora from index");
98  int indexProjects = 0;
99  for(IProject p : root.getProjects()) {

104  log.trace("Finding documents for project {}", p.getName());

106  } catch (CoreException ex) {

108  }
109  monitorDelete.worked(indexProjects++);
110  }
111 
112  SubMonitor monitorImport = SubMonitor.convert(monitor, docList.size());
113  monitorImport.setTaskName("Indexing documents");
114  // index each document
115  int indexDocs = 0;
116  for(Map.Entry<String, URI> e : docList.entries()) {
117  SDocumentGraph docGraph = SaltUtil.loadDocumentGraph(e.getValue());
118 
119  addDocument(e.getKey(), docGraph);
120 
121  monitorImport.worked(indexDocs++);
122  }
123 
124  return Status.OK_STATUS;
125  }
126  };
