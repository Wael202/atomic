97  @Override
98  public void init() {
99  if (getAtomicPepperConfiguration().getPlugInPath() == null) {
100  PepperPropertyException e = new PepperPropertyException("Cannot start Pepper, because no plugin path is given for Pepper modules.");
101  log.error("No plugin path given.", e);
102  throw e;
103  }
104  File pluginPath = new File(getAtomicPepperConfiguration().getPlugInPath());
105  if (!pluginPath.exists()) {
106  PepperOSGiException e = new PepperOSGiException("Cannot load any plugins, since the configured path for plugins '" + pluginPath.getAbsolutePath() + "' does not exist. Please check the entry '" + PepperStarterConfiguration.PROP_PLUGIN_PATH + "' in the Pepper configuration file at '" + getConfiguration().getConfFolder().getAbsolutePath() + "'.");
107  log.error("Plugin path does not exist!", e);
108  throw e;
109  }
110  try {
111  // Disable PepperOSGiRunner and set bundle context
112  System.setProperty(PepperOSGiRunner.PROP_TEST_DISABLED, Boolean.TRUE.toString());
113  System.setProperty(PepperConfiguration.PROP_PEPPER_MODULE_RESOURCES, getAtomicPepperConfiguration().getPlugInPath());
114  setBundleContext(FrameworkUtil.getBundle(this.getClass()).getBundleContext());
115  } catch (Exception e) {
116  log.error("OSGi environment could not be started: {}.", e.getMessage(), e);
117  throw new PepperOSGiException("The OSGi environment could not have been started: " + e.getMessage(), e);
118  }
119  Bundle[] bundles = FrameworkUtil.getBundle(this.getClass()).getBundleContext().getBundles();
120  for (int i = 0; i < bundles.length; i++) {
121  Bundle bundle = bundles[i];
122  String bundleName = bundle.getSymbolicName();
123  if (bundleName != null && bundleName.contains(ARTIFACT_ID_PEPPER_FRAMEWORK)) {
124  frameworkVersion = bundle.getVersion().toString().replace(".SNAPSHOT", "-SNAPSHOT");
125  }
126  }
127 
128  /* Original location of the call
129  * maven = new AtomicMavenAccessor(this);
130  *
131  * Kept for historical reasons.
132  */
133 
134  isInit = true;
135  }

167  @Override

169  if (bundles != null) {
170  Bundle pepperBundle = null;
171  for (Bundle bundle : bundles) {
172  // TODO this is a workaround, to fix that module resolver is

174  // ignored
175  if (PEPPER_FW_MAVEN_QUALIFIED.equalsIgnoreCase(bundle.getSymbolicName())) {
176  pepperBundle = bundle;
177  } else {
178  start(bundle.getBundleId());
179  }
180  }
181  try {
182  if (pepperBundle != null) {
183  pepperBundle.start();
184  }
185  } catch (BundleException e) {
186  throw new PepperOSGiFrameworkPluginException("The Pepper framework bundle could not have been started. Unfortunatly Pepper cannot be started without that OSGi bundle. ", e);
187  }
188  }

250  @Override
251  public Bundle installAndCopy(URI bundleURI) throws BundleException, IOException {
252  // TODO Add logs!
253  Bundle retVal = null;
254  if (bundleURI != null) {
255  String pluginPath = getAtomicPepperConfiguration().getPlugInPath();
256  if (pluginPath != null) {
257  // download file, if file is a web resource
258  if (("http".equalsIgnoreCase(bundleURI.getScheme())) || ("https".equalsIgnoreCase(bundleURI.getScheme()))) {
259  String tempPath = getAtomicPepperConfiguration().getTempPath().getCanonicalPath();
260  URL bundleUrl = bundleURI.toURL();
261  if (!tempPath.endsWith("/")) {
262  tempPath = tempPath + "/";
263  }
264  String baseName = FilenameUtils.getBaseName(bundleUrl.toString());
265  String extension = FilenameUtils.getExtension(bundleUrl.toString());
266  File bundleFile = new File(tempPath + baseName + "." + extension);
267 
268  org.apache.commons.io.FileUtils.copyURLToFile(bundleURI.toURL(), bundleFile);
269  bundleURI = URI.create(bundleFile.getAbsolutePath());
270  }
271  if (bundleURI.getPath().endsWith("zip")) {
272  ZipFile zipFile = null;
273  try {
274  zipFile = new ZipFile(bundleURI.getPath());
275  Enumeration<? extends ZipEntry> entries = zipFile.entries();
276  while (entries.hasMoreElements()) {
277  ZipEntry entry = entries.nextElement();
278  File entryDestination = new File(pluginPath, entry.getName());
279  entryDestination.getParentFile().mkdirs();
280  if (entry.isDirectory()) {
281  entryDestination.mkdirs();
282  } else {
283  InputStream in = zipFile.getInputStream(entry);
284  OutputStream out = new FileOutputStream(entryDestination);
285  IOUtils.copy(in, out);
286  IOUtils.closeQuietly(in);
287  IOUtils.closeQuietly(out);
288  if (entryDestination.getName().endsWith(".jar")) {
289  retVal = install(entryDestination.toURI());
290  }
291  }
292  }
293  } finally {
294  zipFile.close();
295  }
296  } else if (bundleURI.getPath().endsWith("jar")) {
297  File bundleFile = new File(bundleURI.getPath());
298  File jarFile = new File(pluginPath, bundleFile.getName());
299  FileUtils.copyFile(bundleFile, jarFile);
300  retVal = install(jarFile.toURI());
301  }
302  }
303  }
304 
305  return (retVal);
306  }
