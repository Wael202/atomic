236  protected void readDialogSettings() { // NO_UCD (use private)
237  IDialogSettings settings = getDialogSettings();
238  exchangeTargetPath = settings.get(DIALOG_SETTINGS_EXCHANGE_TARGET_PATH);
239  if (exchangeTargetPath == null) {
240  exchangeTargetPath = settings.get(DIALOG_SETTINGS_EXCHANGE_DIRECTORY);
241  }
242 
243  String exchangeTargetTypeName = settings.get(DIALOG_SETTINGS_EXCHANGE_TARGET_TYPE);
244  exchangeTargetType = exchangeTargetTypeName != null ? ExchangeTargetType.valueOf(exchangeTargetTypeName) : ExchangeTargetType.DIRECTORY;
245 
246  String[] modulePropertyKeys = settings.getArray(DIALOG_SETTINGS_MODULE_PROPERTY_KEYS);
247  if (modulePropertyKeys != null && 0 < modulePropertyKeys.length) {
248  String[] modulePropertyValues = settings.getArray(DIALOG_SETTINGS_MODULE_PROPERTY_VALUES);
249  if (modulePropertyValues != null && modulePropertyKeys.length == modulePropertyValues.length) {
250  for (int i = 0; i < modulePropertyKeys.length; ++i) {
251  PepperModuleProperty<String> pepperModuleProperty = new PepperModuleProperty<>(modulePropertyKeys[i], String.class, "", "");
252  pepperModuleProperty.setValueString(modulePropertyValues[i]);
253  addPepperModuleProperty(pepperModuleProperty);
254  }
255  }
256  }

263  protected void writeDialogSettings() {
264  IDialogSettings settings = getDialogSettings();
265 
266  settings.put(DIALOG_SETTINGS_EXCHANGE_TARGET_PATH, exchangeTargetPath);
267  settings.put(DIALOG_SETTINGS_EXCHANGE_TARGET_TYPE, exchangeTargetType.name());
268 
269  PepperModuleDesc module = getPepperModule();
270  settings.put(DIALOG_SETTINGS_MODULE, module != null ? module.getName() : null);
271 
272  FormatDesc fd = getFormatDesc();
273  settings.put(DIALOG_SETTINGS_FORMAT_NAME, fd != null ? fd.getFormatName() : null);
274  settings.put(DIALOG_SETTINGS_FORMAT_VERSION, fd != null ? fd.getFormatVersion() : null);
275 
276  PepperModuleProperties moduleProperties = getPepperModuleProperties();
277  if (moduleProperties != null) {
278  Collection<String> propertyNames = moduleProperties.getPropertyNames();
279  if (propertyNames != null) {
280  String[] modulePropertyKeys = new String[propertyNames.size()];
281  String[] modulePropertyValues = new String[propertyNames.size()];
282  int index = 0;
283  for (String propertyName : propertyNames) {
284  modulePropertyKeys[index] = propertyName;
285  Object value = moduleProperties.getProperty(propertyName).getValue();
286  modulePropertyValues[index] = value != null ? value.toString() : null;
287  ++index;
288  }
289 
290  settings.put(DIALOG_SETTINGS_MODULE_PROPERTY_KEYS, modulePropertyKeys);
291  settings.put(DIALOG_SETTINGS_MODULE_PROPERTY_VALUES, modulePropertyValues);
292  }
293  else {
294  settings.put(DIALOG_SETTINGS_MODULE_PROPERTY_KEYS, (String[]) null);
295  settings.put(DIALOG_SETTINGS_MODULE_PROPERTY_VALUES, (String[]) null);
296  }
297  }
298  else {
299  settings.put(DIALOG_SETTINGS_MODULE_PROPERTY_KEYS, (String[]) null);
300  settings.put(DIALOG_SETTINGS_MODULE_PROPERTY_VALUES, (String[]) null);
301  }
302  }

318  protected List<PepperModuleDesc> getPepperModules(MODULE_TYPE moduleType) {
319  if (pepperModuleDescriptions == null) {
320  pepperModuleDescriptions = new ArrayList<>();
321  }
322  if (!pepperModuleDescriptions.isEmpty()) {
323  return pepperModuleDescriptions;
324  }
325  else {
326  LoadPepperModuleRunnable moduleRunnable = new LoadPepperModuleRunnable(getPepper());
327  try {
328  PlatformUI.getWorkbench().getProgressService().run(false, true, moduleRunnable);
329  }
330  catch (InvocationTargetException | InterruptedException e) {
331  log.error("Loading available non-core Pepper modules did not complete successfully!", e);
332  MessageDialog.openError(Display.getCurrent() != null ? Display.getCurrent().getActiveShell() : Display.getDefault().getActiveShell(), "Pepper modules not loaded!", "The available Pepper modules could not be loaded! Only the core modules will be available!");
333  }
334 
335  // Compile list of module description for use in pages
336  if (getPepper() != null) {
337  Collection<PepperModuleDesc> allModuleDescs = getPepper().getRegisteredModules();
338  if (allModuleDescs != null) {
339  for (PepperModuleDesc desc : allModuleDescs) {
340  if (desc.getModuleType() == moduleType) {
341  pepperModuleDescriptions.add(desc);
342  }
343  }
344  }
345  }
346  if (pepperModuleDescriptions.isEmpty()) {
347  new MessageDialog(this.getShell(), "Error", null, "Did not find any Pepper module of type " + wizardMode.name() + "!", MessageDialog.ERROR, new String[] { IDialogConstants.OK_LABEL }, 0).open();
348  }
349  // Sort list of module descriptions
350  Collections.sort(pepperModuleDescriptions, new Comparator<PepperModuleDesc>() {
351  @Override
352  public int compare(PepperModuleDesc o1, PepperModuleDesc o2) {
353  return o1.getName().compareTo(o2.getName());
354  }
355  });
356  return pepperModuleDescriptions;
357  }
358  }
