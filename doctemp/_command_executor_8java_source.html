112  public void enterAnnotateCommand(AnnotateCommandContext ctx) {
113  if(checkValidEditor()) {
114  List<SNode> elementsToAnnotate = new LinkedList<>();
115 
116  for(Token element : ctx.elements) {
117  SNode newNode = getNodeByID(element.getText());
118  if(newNode != null) {
119  elementsToAnnotate.add(newNode);
120  }
121  }
122 
123  // get the annotation information
124  String ns = ctx.anno.ns == null ? null : ctx.anno.ns.getText();
125  String name = ctx.anno.name.getText();
126 
127  if(ns == null && name != null) {
128  // find first matching namespace of existing annotation
129  ns = "atomic";
130  eachElement:
131  for(SNode n : elementsToAnnotate) {
132  for(SAnnotation anno : n.getAnnotations()) {
133  if(name.equals(anno.getName())) {
134  ns = anno.getNamespace();
135  break eachElement;
136  }
137  }
138  }
139  }
140  String value = ctx.value == null ? null : ctx.value.getText();
141 
142 
143  // add the annotation to each node in the list
144  for(SNode n : elementsToAnnotate) {
145  SAnnotation existing = n.getAnnotation(ns, name);
146  if(existing == null && value != null) {
147  n.createAnnotation(ns, name, value);
148  } else {
149  if(value == null) {
150  // delete the annotation
151  n.removeLabel(ns, name);
152  } else {
153  // change annotation value
154  existing.setValue(value);
155  }
156  }
157  }
158  updateEditor();
159  }

212  public void enterAddOrDeleteEdgeCommand(AddOrDeleteEdgeCommandContext ctx) {

215  SNode sourceNode = getNodeByID(ctx.source.getText());
216  SNode targetNode = getNodeByID(ctx.target.getText());
217  if(sourceNode != null && targetNode != null) {
218 
219  String ns = ctx.anno.ns == null ? null : ctx.anno.ns.getText();
220  String name = ctx.anno.name.getText();
221  String value = ctx.value == null ? null : ctx.value.getText();
222 
223  SALT_TYPE relType = SALT_TYPE.SPOINTING_RELATION;
224  switch(ctx.type.getText()) {
225  case "-d":
226  relType = SALT_TYPE.SDOMINANCE_RELATION;
227  break;
228  case "-p":
229  relType = SALT_TYPE.SPOINTING_RELATION;
230  break;
231  case "-r":
232  relType = SALT_TYPE.SSPANNING_RELATION;
233  break;
234  case "-o":
235  relType = SALT_TYPE.SORDER_RELATION;
236  break;
237  }
238 
239  if(value == null) {
240  // delete all edges of the specified type and with a matching annotation between both nodes
241  List<SRelation> outRels = new LinkedList<>(sourceNode.getOutRelations());
242 
243  for(SRelation rel : outRels) {
244  if(rel.getTarget() == targetNode && relType.getJavaType().isInstance(rel)) {
245  boolean hasAnno = false;
246  for(SAnnotation a : rel.getAnnotations()) {
247  if(name.equals(a.getName()) && (ns == null || ns.equals(a.getNamespace()))) {
248  hasAnno = true;
249  }
250  }
251  if(hasAnno) {
252  graph.removeRelation(rel);
253  }
254  }
255  }
256  } else {
257  graph.createRelation(sourceNode, targetNode, relType, SaltUtil.createQName(ns, name) + "=" + value);
258  }
259  updateEditor();
260  }
