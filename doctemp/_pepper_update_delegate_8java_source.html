115  private IStatus update(IProgressMonitor monitor) {
116  try {
117  if (moduleTable != null) {
118  // Do nothing
119  }
120  else {
121  moduleTable = new HashMap<>();
122  SAXParser saxParser = SAXParserFactory.newInstance().newSAXParser();
123  saxParser.parse(MODULES_XML_PATH, new ModuleTableReader(moduleTable));
124  }
125  }
126  catch (ParserConfigurationException | SAXException | IOException e) {
127  log.error("Getting the Pepper module table from the file at {} didn't succeed!", MODULES_XML_PATH, e);
128  return new PepperUpdateErrorStatus(e);
129  }
130  int workUnits = moduleTable.entrySet().size();
131  SubMonitor subMonitor = SubMonitor.convert(monitor, workUnits);
132  for (Map.Entry<String, Pair<String, String>> entry : moduleTable.entrySet()) {
133  subMonitor.setTaskName("Updating " + entry.getKey() + " ...");
134  boolean cancelled = updateModule(entry, subMonitor.newChild(1));
135  if (cancelled) {
136  log.warn("Update process was cancelled, returning {}.", Status.CANCEL_STATUS);
137  return Status.CANCEL_STATUS;
138  }
139  }
140  return Status.OK_STATUS;
141  }
142 
149  private boolean updateModule(Entry<String, Pair<String, String>> entry, SubMonitor newChild) {
150  if (newChild.isCanceled()) {
151  log.warn("Update was cancelled during processing module {}.", entry.getKey());
152  return true;
153  }
154  AtomicPepperOSGiConnector pepper = (AtomicPepperOSGiConnector) getPepper();
155  if (pepper.update(entry.getValue().getLeft(), entry.getKey(), entry.getValue().getRight(), false, false)) {
156  log.info(entry.getKey().concat(" successfully updated."));
157  resultLines.add(entry.getKey().concat(" successfully updated."));
158  }
159  else {
160  log.info(entry.getKey().concat(" NOT updated."));
161  resultLines.add(entry.getKey().concat(" NOT updated."));
162  }
163  newChild.done();
164  return false;
165  }

254  @Override
255  public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
256  localName = qName.substring(qName.lastIndexOf(":") + 1);
257  if (TAG_LIST.equals(localName)) {
258  defaultRepository = attributes.getValue(ATT_DEFAULTREPO);
259  defaultGroupId = attributes.getValue(ATT_DEFAULTGROUPID);
260  }
261  chars.delete(0, chars.length());
262  }
263 
264  @Override
265  public void characters(char[] ch, int start, int length) throws SAXException {
266  for (int i = start; i < start + length; i++) {
267  chars.append(ch[i]);
268  }
269  }
270 
271  @Override
272  public void endElement(java.lang.String uri, String localName, String qName) throws SAXException {
273  localName = qName.substring(qName.lastIndexOf(":") + 1);
274  if (TAG_ARTIFACTID.equals(localName)) {
275  artifactId = chars.toString();
276  chars.delete(0, chars.length());
277  }
278  else if (TAG_GROUPID.equals(localName)) {
279  groupId = chars.toString();
280  chars.delete(0, chars.length());
281  }
282  else if (TAG_REPO.equals(localName)) {
283  repo = chars.toString();
284  chars.delete(0, chars.length());
285  }
286  else if (TAG_ITEM.equals(localName)) {
287  groupId = groupId == null ? defaultGroupId : groupId;
288  listedModules.put(artifactId, Pair.of(groupId, (repo == null || repo.isEmpty() ? defaultRepository : repo)));
289  chars.delete(0, chars.length());
290  groupId = null;
291  artifactId = null;
292  repo = null;
293  }
294  }
