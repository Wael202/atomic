1013  private static <K, V> List<Map.Entry<K, V>> sortByKey(Map<K, V> map) {
1014  List<Map.Entry<K, V>> entries = new ArrayList<Map.Entry<K, V>>(map.size());
1015 
1016  // don't use addAll with a view of the entries:
1017  // http://findbugs.sourceforge.net/bugDescriptions.html#DMI_ENTRY_SETS_MAY_REUSE_ENTRY_OBJECTS
1018  for (Map.Entry<K, V> e : map.entrySet()) {
1019  entries.add(e);

1021

1023  public int compare(Map.Entry<K, V> e1, Map.Entry<K, V> e2) {
1024  return e1.getKey().toString().compareToIgnoreCase(e2.getKey().toString());
1025 
1026  }
1027  };
1028  // sort key values lexicographically
1029  Collections.sort(entries, comparator);
1030 
1031  return entries;

1320  @Override

1324  if (traversalId.equals(TRAV_MODE_CALC_LEVEL)) {
1325  if (!(relation instanceof SPointingRelation) && !(fromNode instanceof SToken)
1326  && (exportFilter == null || exportFilter.includeNode(currNode)) && (relation != null)) {
1327  currHeight--;

1331 
1332  else if (traversalId.equals(TRAV_MODE_READ_NODES)) {
1333  if (currNode instanceof SToken) {
1334  currHeightFromToken = 1;
1335  // write SSpan-Nodes
1336  if ((fromNode instanceof SSpan) && (!readSpanNodes.contains(fromNode))
1337  && (exportFilter == null || exportFilter.includeNode(fromNode))) {
1338

1340  Set<SAnnotation> sAnnotations = fromNode.getAnnotations();
1341

1343  List<Map.Entry<String, String>> sortedAnnotations = sortAnnotations(sAnnotations);
1344  // use first annotation
1345  annotation = sortedAnnotations.iterator().next().getKey();
1346  }

1349

1352  spanClasses.put(annotation, maxSpanOffset);
1353  }

1355

1357  writeJsonNode(fromNode, maxLevel - currHeightFromToken - spanOffset);
1358  nNodes++;

1360  throw new SaltException("A problem occurred while building JSON objects.");
1361  }

1363 
1364  }
1365 
1366  } else if (currNode instanceof SStructure) {
1367 
1368  if ((exportFilter == null || exportFilter.includeNode(currNode))
1369  && !readStructNodes.contains(currNode)) {
1370  currHeightFromToken++;
1371  int currLevel = maxLevel - currHeightFromToken - spanClasses.size() + 1;
1372  try {
1373  // if currNode is a root
1374  if (roots.contains(currNode)) {
1375  int minRootLevel;
1376

1378  minRootLevel = Math.min(rootToMinLevel.get(currNode), currLevel);

1384  writeJsonNode(currNode, minRootLevel);

1387 
1388  }
1389 
1390  } catch (IOException e) {
1391  throw new SaltException("A problem occurred while building JSON objects.");
1392  }

1394  readStructNodes.add(currNode);
1395  }
1396

1398  if (fromNode instanceof SStructure && roots.contains(fromNode)
1399  && !readStructNodes.contains(fromNode)
1400  && (exportFilter == null || exportFilter.includeNode(fromNode))) {
1401  int thisRootLevel = maxLevel - currHeightFromToken - spanClasses.size();
1402  if (rootToMinLevel.containsKey(fromNode)) {
1403  int oldLevel = rootToMinLevel.get(fromNode);
1404  rootToMinLevel.put(fromNode, Math.min(oldLevel, thisRootLevel));
1405  } else {
1406  rootToMinLevel.put(fromNode, thisRootLevel);
1407  }
1408  }
1409 
1410  }
1411 
1412  if (relation != null) {
1413  if (!readRelations.contains(relation)
1414  && (exportFilter == null || (exportFilter.includeRelation(relation)
1415  && exportFilter.includeNode(fromNode) && exportFilter.includeNode(currNode)))) {
1416  try {
