95  @Override
96  public void postWindowOpen() {
97  log.trace("Hiding basic generic wizards (New Project, Import, Export, etc.).");
98  // Hide basic wizards, e.g., New Project, etc.
99  // FIXME: Try to solve without internal classes!
100  AbstractExtensionWizardRegistry wizardRegistry = (AbstractExtensionWizardRegistry) PlatformUI.getWorkbench().getNewWizardRegistry();
101  IWizardCategory[] categories = PlatformUI.getWorkbench().getNewWizardRegistry().getRootCategory().getCategories();
102  for (IWizardDescriptor wizard : getAllWizards(categories)) {
103  if (wizard.getCategory().getId().matches("org.eclipse.ui.Basic")) {
104  WorkbenchWizardElement wizardElement = (WorkbenchWizardElement) wizard;
105  wizardRegistry.removeExtension(wizardElement.getConfigurationElement().getDeclaringExtension(), new Object[] { wizardElement });
106  }
107  }
108 
109  log.trace("Maximizing application window");
110  getWindowConfigurer().getWindow().getShell().setMaximized(true);
111  }

113  private IWizardDescriptor[] getAllWizards(IWizardCategory[] categories) {
114  List<IWizardDescriptor> results = new ArrayList<IWizardDescriptor>();
115  for (IWizardCategory wizardCategory : categories) {
116  results.addAll(Arrays.asList(wizardCategory.getWizards()));
117  results.addAll(Arrays.asList(getAllWizards(wizardCategory.getCategories())));
118  }
119  return results.toArray(new IWizardDescriptor[0]);
120  }
